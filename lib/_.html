; (union . list) ==> list
;
; Compute the union of a number of sets.
;
; Arguments: a* - sets
;
; Example:   (union '(v w x) '(w x y) '(x y z)) ==> (v w x y z)

(load-from-library "list-to-set.scm")

(define (union . a*)
  (list->set (apply append a*)))
